#Write a function called legendre_symbol(a, p) that calculates the #Legendre symbol​ (a/p), which is a useful function in quadratic #reciprocity. It is defined for an odd #prime p and an integer a not d#ivisible by p as:
  
# (a/p)=1 if a is a quadratic residue modulo p (i.e., there exists and      
  integer x such that x^2 ≡ a (mod p)).

# (a/p)=1 if a is a quadratic residue modulo p.
  You can calculate it using Euler's criterion:(a/p)≡a (p−1)/2 
  (mod p).         


def legendre_symbol(a, p):    
    
    if p <= 2 or p % 2 == 0:
        raise ValueError("p must be an odd prime")
    
    if a % p == 0:
        raise ValueError("a must not be divisible by p")
    
    a = a % p
    
    exponent = (p - 1) // 2
    result = pow(a, exponent, p)  
    if result == 1:
        return 1
    elif result == p - 1:  
        return -1
    else:
        raise ValueError(f"Unexpected result: {result}. p might not be prime.")
